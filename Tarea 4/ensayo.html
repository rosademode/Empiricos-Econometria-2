library(ggfortify)



# Leer los datos desde el archivo Excel
base <- read_excel("pib_fbkf_chile.xlsx", skip = 2)
colnames(base) <- c("Periodo", "fbcf", "PIB")
base$Periodo <- as.Date(base$Periodo, format="%Y-%m-%d")

# Crear variables logartmicas y diferencias logartmicas
base <- base %>%
  mutate(log_PIB = log(PIB), log_fbcf = log(fbcf)) %>%
  mutate(d_log_PIB = log_PIB - lag(log_PIB), d_log_fbcf = log_fbcf - lag(log_fbcf))%>%na.omit()
# Convertir a series de tiempo
log_PIB_ts <- ts(base$log_PIB, start = c(1960, 1), frequency = 12)
log_fbcf_ts <- ts(base$log_fbcf, start = c(1960, 1), frequency = 12)
d_log_PIB_ts <- ts(base$d_log_PIB, start = c(1960, 1), frequency = 12)
d_log_fbcf_ts <- ts(base$d_log_fbcf, start = c(1960, 1),frequency=12)
# Visualizar las series de tiempo
autoplot(log_PIB_ts, main = "Log(PIB)")

autoplot(d_log_PIB_ts, main = "Diferencia Logartmica del PIB")

autoplot(log_fbcf_ts, main = "Log(FBCF)")

autoplot(d_log_fbcf_ts, main = "Diferencia Logartmica del FBCF")

# Calcular y graficar las funciones de autocorrelacin
acf(log_PIB_ts, main = "Funcin de Autocorrelacin del Log(PIB)")

acf(d_log_PIB_ts, main = "Funcin de Autocorrelacin de la Diferencia Logartmica del PIB")

acf(log_fbcf_ts, main = "Funcin de Autocorrelacin del Log(FBCF)")

acf(d_log_fbcf_ts, main = "Funcin de Autocorrelacin de la Diferencia Logartmica del FBCF")
